
```jsx
import { QueryClientProvider } from "@tanstack/react-query";
import { ReactQueryDevtools } from "@tanstack/react-query-devtools";

export default function Providers({ children }) {
  const [queryClient] = useState(() => new QueryClient());
  return (
    <QueryClientProvider client={queryClient}>
      <ReactQueryDevtools initialIsOpen={false} />
      {children}
    </QueryClientProvider>
  );
}
```

Tanstack Query를 사용하려면, 먼저 쿼리 클라이언트와 프로바이더 설정이 필요하다.
왜냐하면 TQ는 `서버 상태` 관리이자, `전역 상태` 관리 도구이기 때문이다.

프로바이더로 감싼 컨택스트 안의 컴포넌트라면, `useQueryClient()`로 쿼리 클라이언트를 언제 어디서든 사용할 수 있다. 

`ReactQueryDevtools`는 해당 클라이언트의 캐시 내용을 볼 수 있는 개발자용 툴을 띄우는 컴포넌트다. (페이지 우하단에 야자수 모양으로 보여준다.)

이건 프로젝트에서 사용한 쿼리 클라이언트 프로바이더인데,
클라이언트  생성 시 `defaultOptions`에서 `stale 타임`, `가비지 타임`, `retry 횟수` 등을 정의할 수 있다.

```jsx
'use client';

import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { ReactQueryDevtools } from '@tanstack/react-query-devtools';
import { useEffect, useState } from 'react';
import { useAuthStore } from '@/shared/store/authStore';

const Providers = ({ children }: { children: React.ReactNode }) => {
  const initAuth = useAuthStore((state) => state.initAuth);
  const [queryClient] = useState(
    () =>
      new QueryClient({
        defaultOptions: {
          queries: {
            staleTime: 1000 * 60 * 5, // 5분
            gcTime: 1000 * 60 * 30, // 30분 (구 cacheTime)
            refetchOnWindowFocus: false,
            retry: 1,
          },
          mutations: {
            retry: 1,
          },
        },
      }),
  );

  useEffect(() => {
    initAuth();
  }, [initAuth]);

  return (
    <QueryClientProvider client={queryClient}>
      {children}
      <ReactQueryDevtools initialIsOpen={false} />
    </QueryClientProvider>
  );
};

export default Providers;
```
