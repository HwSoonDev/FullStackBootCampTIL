- 명시적으로 타입을 지정하지 않아도 자동으로 타입을 추론하는 기능
- 코드를 더 간결하게 작성할 수 있게 해줌
- but 추론이 모호하거나 복잡할 때는 명시적 타입 지정을 권장!
- 
```ts
// 아래 2가지 경우에는 타입 추론 사용 권장.

// 1. 변수 초기화 시 타입 추론
let message = "안녕하세요";           // string으로 추론
let count = 42;                     // number로 추론

// 2. 배열메서드 사용 시 타입 추론
const users: User[] = [
  { id: 1, name: "김개발", age: 25, department: "개발팀" },
  { id: 2, name: "이디자인", age: 30, department: "디자인팀" },
  { id: 3, name: "박기획", age: 28, department: "기획팀" }
];

let userNames = users.map(user => user.name);


// 함수 타입의 반환 타입 추론 🤔 -> 가급적 리턴 타입도 명시하는 것을 권장
function add(a: number, b: number) {
  return a + b;  // number로 반환 타입 추론
}
```
